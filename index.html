  <button onclick="showScreen('home')">Home</button>

  <button onclick="showScreen('requests')">Requests</button>

</div>


<script>

let items = [

  {name:"Lawn Mower", price:8},

  {name:"Power Washer", price:10}

];


let requests = [];


function showScreen(screen) {

  document.querySelectorAll(".screen").forEach(s => s.classList.remove("active"));

  document.getElementById(screen).classList.add("active");

  if(screen === "home") renderHome();

  if(screen === "requests") renderRequests();

}


function renderHome() {

  let html = "<h3>Available Items</h3>";

  items.forEach((item, i) => {

    html += `

      <div class="card">

        <h4>${item.name}</h4>

        <p>$${item.price}/day</p>

        <button onclick="openRent(${i})">Rent</button>

      </div>

    `;

  });

  document.getElementById("home").innerHTML = html;

}


function addItem() {

  let name = document.getElementById("itemName").value;

  let price = document.getElementById("itemPrice").value;


  if(name && price){

    items.push({name, price});

    document.getElementById("itemName").value = "";

    document.getElementById("itemPrice").value = "";

    showScreen("home");

  }

}


function openRent(index) {

  let item = items[index];


  let html = `

    <h3>${item.name}</h3>

    <p>$${item.price}/day</p>


    <label>Days:</label>

    <input id="days" type="number" value="1">


    <label>Payment Method:</label>

    <select id="payment">

      <option value="Card">Card</option>

      <option value="Cash">Cash</option>

    </select>


    <button onclick="submitRequest(${index})">Confirm Request</button>

    <button onclick="showScreen('home')">Back</button>

  `;


  document.getElementById("rentScreen").innerHTML = html;

  showScreen("rentScreen");

}


function submitRequest(index) {

  let days = document.getElementById("days").value;

  let payment = document.getElementById("payment").value;

  let item = items[index];


  let total = item.price * days;


  requests.push({

    item: item.name,

    days: days,

    payment: payment,

    total: total

  });


  alert("Request Sent! Total: $" + total);


  showScreen("requests");

}


function renderRequests() {

  let html = "";


  if(requests.length === 0){

    html = "<p>No requests yet</p>";

  } else {

    requests.forEach(r => {

      html += `

        <div class="card">

          <h4>${r.item}</h4>

          <p>Days: ${r.days}</p>

          <p>Payment: ${r.payment}</p>

          <p>Total: $${r.total}</p>

        </div>

      `;

    });

  }


  document.getElementById("requestList").innerHTML = html;

}


renderHome();
